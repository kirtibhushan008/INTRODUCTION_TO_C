4.CONDITIONAL STATEMENT
#include <stdio.h>
int main()
{
    int num;
    printf("Enter a number:");
    scanf("%d",&num);
    if(num%2==0)
    {
        printf("The given %d is even.",num);
    }
    else
    {
        printf("The given %d is odd.",num);
    }
    return 0;
}
OUTPUT:
Enter a number:5
The given 5 is odd.
