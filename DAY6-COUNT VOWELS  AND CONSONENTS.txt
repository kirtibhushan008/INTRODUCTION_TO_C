#include<stdio.h>
#include<string.h>
#include<ctype.h>
int main(){
char str[200];
int vowels=0,consonants=0,i;
printf("Enter a string: ");
scanf("%[^\n]",str);
for(i=0;str[i]!='\0';i++){
    char ch= tolower(str[i]);
    if(ch>='a' && ch<='z'){
        if(ch=='a' || ch=='e' || ch=='i' || ch=='o' || ch=='u')
        vowels++;
        else
        consonants++;
    }
}
printf("vowels: %d\n",vowels);
printf("consonants: %d\n",consonants);
return 0;
}
OUTPUT:
Enter a string: kartoz
vowels: 2
consonants: 4
