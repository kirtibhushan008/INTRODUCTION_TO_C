#include <stdio.h>
int main() {
    char str[200];
    int i = 0, wordCount = 0;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    while (str[i] == ' ') {
        i++;
    }
    while (str[i] != '\0') {
        if (str[i] != ' ' && (str[i + 1] == ' ' || str[i + 1] == '\0')) {
            wordCount++;
        }
        i++;
    }
    printf("Total number of words: %d\n", wordCount);
    return 0;
}
