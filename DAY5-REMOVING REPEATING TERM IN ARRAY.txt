#include <stdio.h>

int main() {
    int arr[] = {1, 2, 2, 3, 4, 4, 5};
    int n = sizeof(arr) / sizeof(arr[0]);
    int temp[n];
    int i, j, k = 0;
    int isDuplicate;

    for (i = 0; i < n; i++) {
        isDuplicate = 0;
        for (j = 0; j < k; j++) {
            if (arr[i] == temp[j]) {
                isDuplicate = 1;
                break;
            }
        }
        if (!isDuplicate) {
            temp[k] = arr[i];
            k++;
        }
    }
    printf("Array after removing duplicates:\n");
    for (i = 0; i < k; i++) {
        printf("%d ", temp[i]);
    }

    return 0;
}
