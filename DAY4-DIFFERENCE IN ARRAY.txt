#include <stdio.h>

int main() {
    int i, j, k;
    int arr[3][3][3];
    int result;

    printf("Enter 27 elements for the 3x3x3 array:\n");
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++) {
            for (k = 0; k < 3; k++) {
                scanf("%d", &arr[i][j][k]);
            }
        }
    }


    result = arr[0][0][0];

  
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++) {
            for (k = 0; k < 3; k++) {
                
                if (i == 0 && j == 0 && k == 0) continue;
                result -= arr[i][j][k];
            }
        }
    }

    printf("Result of subtracting all elements from the first element = %d\n", result);

    return 0;
}
